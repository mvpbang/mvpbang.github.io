memcache是一款高性能kv缓存系统，多用于缓存高频率查询的结果，不过慢慢被redis取代

env

  • centos7.x 2c4g
  • memcached-v1.5.12

#src

  • https://github.com/libevent/libevent
  • https://github.com/memcached/memcached

1.src-libevent && src-memcache

1.1download

  • libevent-release-1.4.14-stable.tar.gz
  • memcached-1.5.12.tar.gz

1.2requirement

yum install -y autoconf automake libtool  gcc gcc-c++ make

1.3src-binlding libevent

//自动编译生产configure 文件
sh autogen.sh

//构建
./configure --prefix=/usr/local/libevent -q
make -s && make install -s

1.4src-building memcache

./configure --prefix=/servyou/memcached/memcache --with-libevent=/servyou/memcached/libevent  -q
make -s && make install -s

1.5testing

./bin/memcached -d -u weblogic -l 10.111.0.2 -m 200  -p 11211 -c 102400 -U 0  -P /tmp/memcached.pid
