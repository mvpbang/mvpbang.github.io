Vagrant.configure("2") do |config|
    # 定义虚拟机数量（支持批量）
    vms = Array(1..1)
    vms.each do |i|
        config.vm.define "centos8-5#{i}" do |node|
            # 设置虚拟机的Box
            node.vm.box = "centos8"
            # 设置虚拟机的主机名
            node.vm.hostname="centos8-5#{i}"
            # 磁盘容量
            node.vm.disk :disk, size: "40GB", primary: true
            # 设置虚拟机的IP
            node.vm.network "private_network", ip: "172.24.20.5#{i}", hostname: true
            node.vm.provider "virtualbox" do |v|
                # 设置虚拟机的名称
                v.name = "centos8-20.5#{i}"
                # 设置虚拟机的内存大小  
                v.memory = 4096
                # 设置虚拟机的CPU个数
                v.cpus = 2
            end
        end
        # 自定义执行脚本
        config.vm.provision "shell", path: "resizedisk.sh", run: "once"
        config.vm.provision "shell", path: "init.sh", run: "once"
    end
end

#!/bin/sh

sudo su - <<-'EOG'

# enable passwd auth and reset root passwd
sed -i.bak 's#PasswordAuthentication no#PasswordAuthentication yes#' /etc/ssh/sshd_config
systemctl restart sshd
echo root:321321 |chpasswd

# disable selinux
sed -i.bak 's#SELINUX=enforcing#SELINUX=disabled#' /etc/selinux/config

# disable firwalld
systemctl mask firewalld

# china repo
rm -rf /etc/yum.repos.d/*  && curl -o /etc/yum.repos.d/CentOS-Base.repo https://mirrors.aliyun.com/repo/Centos-vault-8.5.2111.repo &&curl -o /etc/yum.repos.d/epel-archive-8.repo  https://mirrors.aliyun.com/repo/epel-archive-8.repo

dnf clean all && dnf makecache

# install requirement
dnf install -y vim lrzsz wget tree unzip
EOG
