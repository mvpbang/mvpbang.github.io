3.githu setting

  • create repository
  • clone && push hugo website
  • settings –>github pages

#baseurl

config.toml
baseURL = "https://mvpbang.com/"

#output static pages

hugo server -D
hugo -d docs

#push

git push xxxx

#github pages 

github pages
   main  /docs
   mvpbang.com   //custom domain
   enable https  //wait minitus

#domain

ping xxxx.github.io   //extract ip

#domain A/CNAME

@   A  ip
www CNAME xxx.github.com
