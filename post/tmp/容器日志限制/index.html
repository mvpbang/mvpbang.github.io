/etc/docker/daemon.json
{
    "registry-mirrors": [
        "http://f613ce8f.m.daocloud.io"
    ], 
    "log-driver": "json-file", 
    "log-opts": {
        "max-size": "100m", 
        "max-file": "3"
    }
}

systemctl daemon-reload  && systemctl restart docker

注意这个修改只对新容器生效，老容器不生效
